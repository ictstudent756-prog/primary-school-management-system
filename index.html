---------------------------- # Models # ---------------------------- class Student(models.Model): user = models.OneToOneField(User, on_delete=models.CASCADE) grade = models.CharField(max_length=10) parent_contact = models.CharField(max_length=15) def __str__(self): return self.user.username class Teacher(models.Model): user = models.OneToOneField(User, on_delete=models.CASCADE) subject = models.CharField(max_length=50) def __str__(self): return self.user.username class ReportCard(models.Model): student = models.ForeignKey(Student, on_delete=models.CASCADE) subject = models.CharField(max_length=50) grade = models.CharField(max_length=2) remarks = models.TextField() def __str__(self): return f"{self.student.user.username} - {self.subject}" # ---------------------------- # Views # ---------------------------- def is_teacher(user): return hasattr(user, 'teacher') def is_student(user): return hasattr(user, 'student') @login_required def dashboard(request): if is_teacher(request.user): return render(request, "teacher_dashboard.html") elif is_student(request.user): return render(request, "student_dashboard.html") else: return render(request, "admin_dashboard.html") @login_required def certificate(request, student_id): student = Student.objects.get(id=student_id) response = HttpResponse(content_type='application/pdf') response['Content-Disposition'] = f'attachment; filename="{student.user.username}_certificate.pdf"' p = canvas.Canvas(response) p.drawString(100, 750, "Certificate of Completion") p.drawString(100, 700, f"This certifies that {student.user.get_full_name()} has completed Grade {student.grade}.") p.showPage() p.save() return response # ---------------------------- # URL Configuration # ---------------------------- urlpatterns = [ path('admin/', admin.site.urls), path('dashboard/', dashboard, name='dashboard'), path('certificate//', certificate, name='certificate'), ] #
